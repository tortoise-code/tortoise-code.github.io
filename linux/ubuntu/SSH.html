# Ubuntu ssh远程连接

# ssh-server 服务端 用来被远程连接

# ssh-client 客户端 用来远程链接服务器 ubuntu默认自带

ssh username@192.168.1.103 # 链接
exit                       # 断开链接

# UBUNTU client 与 server 传输文件

  • 从远处复制文件到本地目录
scp pi@192.168.1.178:/home/pi/tt.txt ~/test
  • 从远处复制目录到本地
scp pi@192.168.1.178:/home/pi/mongodb ~/test
  • 上传本地文件到远程机器指定目录
scp ~/test/a.out pi@192.168.1.178:/home/pi

#  主目录创建svn仓库
mkdir svn
#  进入目录 创建 文件夹 log  project  conf
cd svn
mkdir log project
# 进入真实的svn仓库
cd project
# 创建项目仓库 (宝塔面板风格)
svnadmin create www.yourdomain.com
# 将 仓库配置文件夹移动到svn下成为全局配置
mv ~/svn/project/www.yourdomain.com/conf/  ~/svn/
